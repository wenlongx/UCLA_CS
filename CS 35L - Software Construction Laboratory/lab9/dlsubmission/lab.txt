    #!/bin/bash
    while IFS='' read -r line || [[ -n "$line" ]]; do
        eval "ldd /usr/bin/$line" >&1
    done < "$1"

    chmod +x ldd-commands.sh
    ./ldd-commands.sh commands.txt > temp
    sort -u temp > sorted.txt
    sed 's/^.* => //g' sorted.txt > temp.txt
